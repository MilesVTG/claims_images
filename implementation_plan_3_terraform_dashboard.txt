### 2. Full BigQuery View for Claims Team (Ready-to-Run)

Run this SQL in BigQuery once (it creates a clean, easy-to-use view):

```sql
CREATE OR REPLACE VIEW `fraud_detection.claims_team_dashboard` AS
SELECT
  c.contract_id,
  c.claim_id,
  c.claim_date,
  c.reported_loss_date,
  c.service_drive_location,
  c.service_drive_coords,

  -- Gemini Analysis (flattened)
  c.risk_score,
  ARRAY_TO_STRING(c.red_flags, ' | ') AS red_flags_list,
  JSON_VALUE(c.gemini_analysis, '$.explanation') AS gemini_explanation,
  JSON_VALUE(c.gemini_analysis, '$.recommendation') AS recommendation,

  -- Auto-detected changes (from Gemini JSON)
  JSON_VALUE(c.gemini_analysis, '$.red_flags[0]') LIKE '%tire%'
    OR JSON_VALUE(c.gemini_analysis, '$.red_flags[1]') LIKE '%tire%' AS tire_brand_changed,
  JSON_VALUE(c.gemini_analysis, '$.red_flags[0]') LIKE '%color%'
    OR JSON_VALUE(c.gemini_analysis, '$.red_flags[1]') LIKE '%color%' AS vehicle_color_changed,

  -- Reverse Image Results (flattened)
  JSON_VALUE(c.reverse_image_results, '$.full_matching_images[0]') IS NOT NULL AS has_exact_web_match,
  ARRAY_LENGTH(JSON_QUERY_ARRAY(c.reverse_image_results, '$.full_matching_images')) AS full_match_count,
  ARRAY_LENGTH(JSON_QUERY_ARRAY(c.reverse_image_results, '$.visually_similar_images')) AS similar_images_count,
  JSON_VALUE(c.reverse_image_results, '$.pages_with_matching_images[0]') AS first_matching_page,

  -- Metadata
  JSON_VALUE(c.extracted_metadata, '$.DateTimeOriginal') AS photo_timestamp,
  JSON_VALUE(c.extracted_metadata, '$.gps_location') AS photo_gps,
  c.photo_uris[SAFE_OFFSET(0)] AS latest_photo_uri,
  c.processed_at,

  -- Historical context for this contract (last 3 claims)
  ARRAY_AGG(
    STRUCT(
      prev.claim_date AS prev_claim_date,
      prev.risk_score AS prev_risk_score
    )
    ORDER BY prev.claim_date DESC
  ) OVER (PARTITION BY c.contract_id ORDER BY c.claim_date ROWS BETWEEN 3 PRECEDING AND 1 PRECEDING) AS recent_history

FROM `fraud_detection.claims` c
LEFT JOIN `fraud_detection.claims` prev
  ON c.contract_id = prev.contract_id
  AND prev.claim_date < c.claim_date
ORDER BY c.contract_id, c.claim_date DESC;
```
