### Updated main.py (now with automatic reverse image lookup)
```python
import json
import io
from datetime import datetime
from PIL import Image, ExifTags
from google.cloud import storage, bigquery, vision

def extract_exif_and_reverse_image(event, context):
    """Triggered by new photo upload → EXIF + Google reverse image lookup + BigQuery"""
    bucket_name = event['bucket']
    file_name = event['name']
    gs_uri = f"gs://{bucket_name}/{file_name}"

    # 1. Download image
    storage_client = storage.Client()
    bucket = storage_client.bucket(bucket_name)
    blob = bucket.blob(file_name)
    image_bytes = blob.download_as_bytes()

    # 2. EXIF extraction (same as before, now cleaned)
    img = Image.open(io.BytesIO(image_bytes))
    exif_data = {}
    exif = img.getexif()
    gps = None

    if exif:
        for tag_id, value in exif.items():
            tag = ExifTags.TAGS.get(tag_id, tag_id)
            if tag == "GPSInfo":
                gps = parse_gps(value)
                exif_data["gps_lat"] = gps["lat"]
                exif_data["gps_lon"] = gps["lon"]
                exif_data["gps_location"] = f"{gps['lat']},{gps['lon']}"
            elif isinstance(value, bytes):
                exif_data[tag] = value.decode(errors="ignore")
            else:
                exif_data[tag] = str(value)

    # 3. Google Reverse Image Lookup (Web Detection) — AUTOMATIC
    vision_client = vision.ImageAnnotatorClient()
    image = vision.Image(source=vision.ImageSource(image_uri=gs_uri))

    response = vision_client.annotate_image({
        "image": image,
        "features": [{"type_": vision.Feature.Type.WEB_DETECTION}]
    })

    web_detection = response.web_detection
    reverse_image_results = {
        "full_matching_images": [img.url for img in web_detection.full_matching_images],
        "partial_matching_images": [img.url for img in web_detection.partial_matching_images],
        "visually_similar_images": [img.url for img in web_detection.visually_similar_images],
        "pages_with_matching_images": [page.url for page in web_detection.pages_with_matching_images],
        "web_entities": [{"entity": e.description, "score": e.score} for e in web_detection.web_entities]
    }

    # 4. Insert everything into BigQuery
    bq_client = bigquery.Client()
    table_id = "fraud_detection.claims"

    row = {
        "contract_id": extract_contract_from_filename(file_name),  # ← customize this function
        "claim_id": file_name.split("/")[0],
        "photo_uris": [gs_uri],
        "extracted_metadata": json.dumps(exif_data),
        "reverse_image_results": json.dumps(reverse_image_results),
        "processed_at": datetime.utcnow().isoformat()
    }

    errors = bq_client.insert_rows_json(table_id, [row])
    if errors:
        print(f"BigQuery insert error: {errors}")
    else:
        print(f"✅ Processed {file_name} with reverse image check")

def parse_gps(gps_info):
    def dms_to_dd(dms, ref):
        degrees = dms[0][0] / dms[0][1]
        minutes = dms[1][0] / dms[1][1]
        seconds = dms[2][0] / dms[2][1]
        dd = degrees + minutes/60 + seconds/3600
        return dd if ref in "NE" else -dd
    try:
        lat = dms_to_dd(gps_info[2], gps_info[1])
        lon = dms_to_dd(gps_info[4], gps_info[3])
        return {"lat": lat, "lon": lon}
    except:
        return {"lat": None, "lon": None}

# Helper – customize for your filename format
def extract_contract_from_filename(filename):
    # Example: "contract_ABC123456_claim_001.jpg" → "ABC123456"
    try:
        return filename.split("contract_")[1].split("_claim")[0]
    except:
        return "UNKNOWN"
```

### How to deploy the upgraded function
```bash
gcloud functions deploy photo-fraud-trigger \
  --gen2 \
  --region=us-central1 \
  --runtime=python312 \
  --trigger-bucket=YOUR_PHOTOS_BUCKET \
  --entry-point=extract_exif_and_reverse_image \
  --memory=1024MB \
  --timeout=120s
```
